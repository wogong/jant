Insert 0 as an ellipsis marker for gaps. */ -}} {{- $pages := slice -}} {{- if le $total 7 -}} {{- range seq 1 $total -}} {{- $pages = $pages | append . -}} {{- end -}} {{- else -}} {{- $set := slice 1 $total $current -}} {{- if gt $current 1 -}}{{- $set = $set | append (sub $current 1) -}}{{- end -}} {{- if lt $current $total -}}{{- $set = $set | append (add $current 1) -}}{{- end -}} {{- $sorted := sort (uniq $set) -}} {{- $prev := 0 -}} {{- range $i, $p := $sorted -}} {{- if and (gt $i 0) (gt (sub $p $prev) 1) -}} {{- $pages = $pages | append 0 -}} {{- end -}} {{- $pages = $pages | append $p -}} {{- $prev = $p -}} {{- end -}} {{- end -}}
